interface SearchResult {
  id?: string;
  title: string;
  composer: string;
  textWriter?: string;
  description?: string;
  language?: string;
  voicing?: string;
  difficulty?: string;
  season?: string;
  theme?: string;
  sourceLink?: string;
}
 
interface SearchResultsProps {
  results: SearchResult[];
  query: string;
  hasSearched: boolean;
  isLoading: boolean;
  error: string;
}
 
export const SearchResults = ({ 
  results, 
  query, 
  hasSearched, 
  isLoading, 
  error 
}: SearchResultsProps) => {
  if (error) {
    return (
      <div className="card bg-red-50 border-red-200">
        <div className="flex items-center gap-2 text-red-800">
          <span className="text-red-500">⚠️</span>
          <strong>Error:</strong> {error}
        </div>
      </div>
    );
  }
 
  if (isLoading) {
    return (
      <div className="card text-center">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-primary-600 mx-auto mb-4"></div>
        <p className="text-secondary-600">Searching for music...</p>
      </div>
    );
  }
 
  if (results.length > 0) {
    return (
      <div className="card animate-fade-in">
        <h3 className="text-xl font-semibold text-secondary-800 mb-4">
          Search Results ({results.length} found)
        </h3>
        <div className="space-y-4">
          {results.map((result, index) => (
            <SearchResultItem key={result.id || index} result={result} />
          ))}
        </div>
      </div>
    );
  }
 
  if (hasSearched && results.length === 0) {
    return (
      <div className="card text-center">
        <div className="text-6xl mb-4">🎵</div>
        <h3 className="text-lg font-medium text-secondary-800 mb-2">
          No results found for "{query}"
        </h3>
        <p className="text-secondary-600">
          Try different search terms or check your spelling
        </p>
      </div>
    );
  }
 
  return (
    <div className="card text-center">
      <div className="text-6xl mb-4">🎼</div>
      <h3 className="text-lg font-medium text-secondary-800 mb-2">
        Welcome to Choir Music Search
      </h3>
      <p className="text-secondary-600 mb-4">
        Enter a search term above to find choir music!
      </p>
      <div className="text-sm text-secondary-500">
        <p>Try searching for composers like "Bach", "Mozart", or styles like "Christmas", "Latin"</p>
      </div>
    </div>
  );
};
 
const SearchResultItem = ({ result }: { result: SearchResult }) => {
  const metadata = [
    { label: 'Composer', value: result.composer },
    { label: 'Text Writer', value: result.textWriter },
    { label: 'Description', value: result.description },
    { label: 'Language', value: result.language },
    { label: 'Voicing', value: result.voicing },
    { label: 'Difficulty', value: result.difficulty },
    { label: 'Season', value: result.season },
    { label: 'Theme', value: result.theme },
  ].filter(item => item.value);
 
  return (
    <div className="search-result animate-slide-up">
      <h4 className="text-lg font-semibold text-secondary-800 mb-2">
        {result.title}
      </h4>
      
      <div className="grid grid-cols-1 md:grid-cols-2 gap-2 mb-3">
        {metadata.map((item, index) => (
          <div key={index} className="text-sm">
            <span className="font-medium text-secondary-600">{item.label}:</span>{' '}
            <span className="text-secondary-800">{item.value}</span>
          </div>
        ))}
      </div>
 
      {result.sourceLink && (
        <a
          href={result.sourceLink}
          target="_blank"
          rel="noopener noreferrer"
          className="inline-flex items-center gap-1 text-primary-600 hover:text-primary-700 font-medium transition-colors"
        >
          View Source
          <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M10 6H6a2 2 0 00-2 2v10a2 2 0 002 2h10a2 2 0 002-2v-4M14 4h6m0 0v6m0-6L10 14" />
          </svg>
        </a>
      )}
    </div>
  );
};
